<script>
document.addEventListener('DOMContentLoaded', function() {
    window.scrollToComments = function(event) {
        event.preventDefault();
        document.getElementById('disqus_thread').scrollIntoView({
            behavior: 'smooth'
        });
    }

    window.scrollToTop = function(event) {
        event.preventDefault();
        window.scrollTo({
            top: 0,
            behavior: 'smooth'
        });
    }
});
</script>
